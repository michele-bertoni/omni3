833  /* If there are no scheduled movements, perform the default indefinite movement */
834  if (freeIndex <= 0) {
835  return this->defaultMovement->getSpeed(time, targetSpeed);
836  }
837 
838  /* Compute brakingSpace given current speed and friction coefficients vectors */

841  brakingSpace[STRAFE] = pow2(currentSpeed[STRAFE]) * frictionCoefficient[STRAFE];
842  brakingSpace[THETA] = pow2(currentSpeed[THETA]) * frictionCoefficient[THETA];
843 
844  /* While current movement is finished, shift schedule of one position */
845  while(this->movementsSchedule[0]->isFinished(currentPosition, brakingSpace, time)) {
846  delete movementsSchedule[0];
847  for(int i=0; i<freeIndex-1; i++) {
848  movementsSchedule[i] = movementsSchedule[i+1];
849  }
850  freeIndex--;
851  movementsSchedule[freeIndex] = nullptr;
852  }
853

855  otherwise perform current finite movement */
856  return freeIndex <= 0 ? this->defaultMovement->getSpeed(time, targetSpeed) :
