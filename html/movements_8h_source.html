241  /* If targetTime is 0 (first call of this method on this object), initialize targetTime */
242  if(this->startTime == 0) {
243  /* Since a target time of 0 is seen as uninitialized, if result is by chance 0, it is set to 1 (1 ms of
244  difference is very small, also it is unlikely that the result is 0) */
245  this->startTime = time!=0 ? time : 1;
246  }
247 
248  /* If current duration of the movement is at least the theoretical duration, end current movement */
249  if((time - startTime) >= lround(_duration * TO_MILLIS)) {
250  return true;
251  }
252 
253  /* Compute displacements in [FORWARD, STRAFE, THETA] frame of reference */
254  FiniteMovement::xyToSF(target[POS_X]-position[POS_X], target[POS_Y]-position[POS_Y],
255  position[POS_PHI], &displacements[FORWARD], &displacements[STRAFE]);
256  displacements[THETA] = Movement::angularDistance(target[POS_PHI], position[POS_PHI]);
257

260  this->_isFinished[FORWARD] =
261  (displacements[FORWARD] <= max(brakingSpace[FORWARD], linearTolerance)) &&
262  (displacements[FORWARD] >= -max(brakingSpace[FORWARD], linearTolerance));
263  this->_isFinished[STRAFE] =

265  (displacements[STRAFE] >= -max(brakingSpace[STRAFE], linearTolerance));
266  this->_isFinished[THETA] =

268  (displacements[THETA] >= -max(brakingSpace[THETA], angularTolerance));
