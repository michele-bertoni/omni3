850  /* If there are no scheduled movements, perform the default indefinite movement */
851  if (freeIndex <= 0) {

853  }

858  brakingSpace[STRAFE] = pow2(currentSpeed[STRAFE]) * frictionCoefficient[STRAFE];
859  brakingSpace[THETA] = pow2(currentSpeed[THETA]) * frictionCoefficient[THETA];
860 
861  /* While current movement is finished, shift schedule of one position */
862  while(this->movementsSchedule[0]->isFinished(currentPosition, brakingSpace, time)) {
863  delete movementsSchedule[0];
864  for(int i=0; i<freeIndex-1; i++) {
865  movementsSchedule[i] = movementsSchedule[i+1];
866  }
867  freeIndex--;
868  movementsSchedule[freeIndex] = nullptr;
869  }
870 
871  /* If there are no scheduled movements, perform the default indefinite movement,
872  otherwise perform current finite movement */
873  return freeIndex <= 0 ? this->defaultMovement->getSpeed(time, targetSpeed) :
